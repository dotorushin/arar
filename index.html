<body style="margin: 0; overflow: hidden;">

    <a-scene embedded arjs>

        <a-entity position="0 0 -2">

            <a-gltf-model src="your_model.glb" scale="1 1 1" gesture-handler></a-gltf-model>

        </a-entity>

        <a-camera position="0 0 0"></a-camera>

    </a-scene>


    <script>

        AFRAME.registerComponent('gesture-handler', {

            init: function () {

                let el = this.el;

                let scale = 1;

                let isDragging = false;

                let lastX = 0;

                let lastY = 0;


                document.addEventListener('touchstart', (event) => {

                    if (event.touches.length === 1) {

                        isDragging = true;

                        lastX = event.touches[0].clientX;

                        lastY = event.touches[0].clientY;

                    }

                });


                document.addEventListener('touchmove', (event) => {

                    if (isDragging && event.touches.length === 1) {

                        let deltaX = event.touches[0].clientX - lastX;

                        let deltaY = event.touches[0].clientY - lastY;


                        let currentPosition = el.getAttribute('position');

                        el.setAttribute('position', {

                            x: currentPosition.x + deltaX * 0.01,

                            y: currentPosition.y - deltaY * 0.01,

                            z: currentPosition.z

                        });


                        lastX = event.touches[0].clientX;

                        lastY = event.touches[0].clientY;

                    } else if (event.touches.length === 2) {

                        let touch1 = event.touches[0];

                        let touch2 = event.touches[1];

                        let distance = Math.hypot(

                            touch1.clientX - touch2.clientX,

                            touch1.clientY - touch2.clientY

                        );


                        if (this.lastDistance) {

                            let scaleChange = (distance - this.lastDistance) * 0.005;

                            scale += scaleChange;

                            el.setAttribute('scale', { x: scale, y: scale, z: scale });

                        }

                        this.lastDistance = distance;

                    }

                });


                document.addEventListener('touchend', () => {

                    isDragging = false;

                    this.lastDistance = null;

                });

            }

        });

    </script>

</body>
